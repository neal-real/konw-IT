## 创建文件

- 右键->新建 pages;
- 小程序自动生成的文件中`****.js`



```js
// pages/my/my.js
Page({

  /**
   * 页面的初始数据,在 page 内的 data 属性下的数据都可以直接在 wxml 里面直接使用

   */
  data: {
		name:'初始数据'
  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {

   */
  onReady: function () {

  },

  /**
   * 生命周期函数--监听页面显示
   */
  onShow: function () {

  },

  /**
   * 生命周期函数--监听页面隐藏
   */
  onHide: function () {

  },

  /**
   * 生命周期函数--监听页面卸载
   */
  onUnload: function () {

  },

  /**
   * 页面相关事件处理函数--监听用户下拉动作
   */
  onPullDownRefresh: function () {

  },

  /**
   * 页面上拉触底事件的处理函数
   */
  onReachBottom: function () {

  },

  /**
   * 用户点击右上角分享
   */
  onShareAppMessage: function () {

  }
})
```
