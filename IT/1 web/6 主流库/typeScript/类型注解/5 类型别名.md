



```ts
// 给一个特定的类型限制
type User = { name: string; age: number };
// 数组中的每个值必须是 User 类型
const objectArr: User[] = [
  {
    name: 'dell',
    age: 28
  },
  {
    name: 'dell2',
    age: 18
  },
];

class Teacher {
  name: string;
  age: number;
}
// 数组中的每个值必须是 Teacher 类型
const objectArr: Teacher[] = [
  new Teacher(),
  {
    name: 'dell',
    age: 28
  }
];
```
